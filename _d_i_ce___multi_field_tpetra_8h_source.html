430  scalar_t norm(Teuchos::RCP<MultiField> multifield){
431  TEUCHOS_TEST_FOR_EXCEPTION(this->get_map()->get_num_local_elements()!=
432  multifield->get_map()->get_num_local_elements(),std::runtime_error,
433  "Error, incompatible multifield maps");
434  scalar_t norm = 0.0;
435  for(int_t i=0;i<get_map()->get_num_local_elements();++i){
436  norm += (this->local_value(i)-multifield->local_value(i))*
437  (this->local_value(i)-multifield->local_value(i));
438  }
439  norm = std::sqrt(norm);
440  return norm;
441  }
