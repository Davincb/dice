416  scalar_t norm(Teuchos::RCP<MultiField> multifield){
417  TEUCHOS_TEST_FOR_EXCEPTION(this->get_map()->get_num_local_elements()!=
418  multifield->get_map()->get_num_local_elements(),std::runtime_error,
419  "Error, incompatible multifield maps");
420  scalar_t norm = 0.0;
421  for(int_t i=0;i<get_map()->get_num_local_elements();++i){
422  norm += (this->local_value(i)-multifield->local_value(i))*
423  (this->local_value(i)-multifield->local_value(i));
424  }
425  norm = std::sqrt(norm);
426  return norm;
427  }
