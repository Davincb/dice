386  {
387  if(rank_!=field_enums::INTERNAL_FACE_EDGE_RANK && rank_!=field_enums::INTERNAL_CELL_RANK && rank_!= EXTERNAL_FACE_EDGE_RANK) return false;

389  else return false;
