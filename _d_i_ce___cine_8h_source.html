149 class DICE_LIB_DLL_EXPORT
150 Cine_Header
151 {
152 public:
157  Cine_Header(const std::string & file_name,
158  const cine_file_header & header,
159  const bitmap_info_header & bitmap_header):
160  header_(header),
161  bitmap_header_(bitmap_header),
162  file_name_(file_name),
163  bit_depth_(NO_SUCH_BIT_DEPTH),
164  frame_rate_(0){
165  image_offsets_ = new int64_t[header_.ImageCount];
166  // set the bit depth of the images
167  int_t bit_depth = (bitmap_header_.biSizeImage * 8) / (bitmap_header_.biWidth * bitmap_header_.biHeight);
168  if(bit_depth==8) bit_depth_=BIT_DEPTH_8;
169  else if (bit_depth==16) bit_depth_=BIT_DEPTH_16;
170  else if (bit_depth==10) bit_depth_=BIT_DEPTH_10_PACKED;
171  else{
172  TEUCHOS_TEST_FOR_EXCEPTION(true,std::runtime_error,"Error, invalid bit depth .cine");
173  }
174  };
176  virtual ~Cine_Header(){
177  delete[] image_offsets_;
178  };

182  bitmap_info_header bitmap_header_;
184  int64_t * image_offsets_;
186  std::string file_name_;
188  Bit_Depth bit_depth_;
