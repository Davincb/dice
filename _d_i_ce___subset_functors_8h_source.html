129  scalar_t mean_sum_d_;
139  ZNSSD_Gamma_Functor(intensity_device_view_1d ref_intensities,
140  intensity_device_view_1d def_intensities,
141  bool_device_view_1d is_active,
142  bool_device_view_1d is_deactivated_this_step,
143  const scalar_t & mean_r,
144  const scalar_t & mean_d,
145  const scalar_t & mean_sum_r,
146  const scalar_t & mean_sum_d):
147  ref_intensities_(ref_intensities),
148  def_intensities_(def_intensities),
149  is_active_(is_active),
150  is_deactivated_this_step_(is_deactivated_this_step),
151  mean_r_(mean_r),
152  mean_d_(mean_d),
153  mean_sum_r_(mean_sum_r),
154  mean_sum_d_(mean_sum_d){}
156  KOKKOS_INLINE_FUNCTION
157  void operator()(const int_t pixel_index, scalar_t & gamma) const{
158  if(is_active_(pixel_index)&!is_deactivated_this_step_(pixel_index)){
159  scalar_t value = (def_intensities_(pixel_index)-mean_d_)/mean_sum_d_ - (ref_intensities_(pixel_index)-mean_r_)/mean_sum_r_;
160  //printf("%i value: %e %e %e %e %e %e %e\n",pixel_index,value,def_intensities_(pixel_index),mean_d_,mean_sum_d_,ref_intensities_(pixel_index),mean_r_,mean_sum_r_);
161  gamma += value*value;
162  }
163  }
164 };
