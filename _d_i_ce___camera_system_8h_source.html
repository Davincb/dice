141  scalar_t diff(const Camera_System & rhs) const{
142  TEUCHOS_TEST_FOR_EXCEPTION(rhs.num_cameras()!=num_cameras(),std::runtime_error,"");
143  scalar_t diff_ = 0.0;
144  for(size_t i=0;i<num_cameras();++i)
145  diff_ += camera(i)->diff(*(rhs.camera(i).get()));
